#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int value;
    struct Node* next;
} Node;

typedef struct List {
    Node* head;
    Node* tail;
    int length;
} List;

Node* node_create(int value) {
    Node* newNode = malloc(sizeof(Node));
    newNode->value = value;
    newNode->next = NULL;
    return newNode;
}

List* list_create() {
    List* newList = malloc(sizeof(List));

    newList->head = NULL;
    newList->tail = NULL;
    newList->length = 0;

    return newList;
}

int list_insert_at_end(List *list, int value) {
    Node* newNode = node_create(value);
    if (newNode == NULL) return -1;

    if (list->tail) {
        list->tail->next = newNode;
    } else {
        list->tail = list->head = newNode;
    }

    list->length++;
    return 0;
}

int list_insert_at_start(List *list, int value) {
    Node *newNode = node_create(value);
    if (newNode == NULL) return -1;

    if (list->head) {
        newNode->next = list->head;
        list->head = newNode;
    } else {
        list->head = list->tail = newNode;
    }

    list->length++;
    return 0;
}

int list_insert_at_index(List *list, int value, int index) {
    if (index == 0) return list_insert_at_start(list, value);
    if (index == list->length) return list_insert_at_end(list, value);
    if (index > list->length) return -2;

    Node *newNode = node_create(value);
    if (newNode == NULL) return -1;

    Node *currNode = list->head;
    for (int i = 0; i < index - 1; i++) {
        currNode = currNode->next;
    }

    newNode->next = currNode->next;
    currNode->next = newNode;

    list->length++;
    return 0;
}

int list_remove_from_end(List *list) {
    if (!list->head) return -2;

    Node *currNode = list->head, *prevNode = NULL;
    while (currNode != list->tail) {
        prevNode = currNode;
        currNode = currNode->next;
    }

    if (prevNode) {
        list->tail = prevNode;
        prevNode->next = NULL;
    } else {
        list->tail = list->head = NULL;
    }

    list->length--;
    return currNode->value;
}

int list_remove_from_start(List *list) {
    if (!list->head) return -2;

    Node *node = list->head;
    list->head = node->next;
    if (!list->head) list->tail = NULL;

    list->length--;
    return node->value;
}

int list_remove_from_index(List *list, int index) {
    if (index == 0) return list_remove_from_start(list);
    if (index == list->length-1) return list_remove_from_end(list);
    if (index >= list->length) return -2;

    Node *currNode = list->head, *prevNode = NULL;
    for (int i = 0; i < index; i++) {
        prevNode = currNode;
        currNode = currNode->next;
    }

    prevNode->next = currNode->next;
    list->length--;
    return currNode->value;
}
                        
class Data;

class List {
public:
    void insertAtStart(Data*);
    void insertAtEnd(Data*);
    void insertAtIndex(Data*, int);

    Data* removeFromStart();
    Data* removeFromEnd();
    Data* removeFromIndex(int);

    Data* getFromStart();
    Data* getFromEnd();
    Data* getFromIndex(int);

    bool isEmpty();

    ~List();
private:
    struct Node {
        Data *data;
        Node *next;

        Node(Data *data): data(data), next(nullptr) {}
    };

    Node *head = nullptr;
    Node *tail = nullptr;
};

void List::insertAtStart(Data *data) {
    Node *newNode = new Node(data);

    if (!head) {
        head = tail = newNode;
    } else {
        newNode->next = head;
        head = newNode;
    }
}

void List::insertAtEnd(Data *data) {
    Node *newNode = new Node(data);

    if (!tail) {
        head = tail = newNode;
    } else {
        tail->next = newNode;
    }
}

void List::insertAtIndex(Data *data, int index) {
    if (index < 0) return;

    Node *newNode = new Node(data);

    Node *curr = head, *prev = nullptr;
    int i = 0;
    while (i < index and curr != nullptr) {
        prev = curr;
        curr = curr->next;
        i++;
    }

    if (prev == nullptr) {
        newNode->next = head;
        head = newNode;

        if (tail == nullptr) tail = head;
    } else {
        prev->next = newNode;
        newNode->next = curr;

        if (tail == prev) tail = newNode;
    }
}

Data *List::removeFromStart() {
    if (head == nullptr) return nullptr;

    Node *node = head;
    head = head->next;
    if (head == nullptr) tail = nullptr;

    Data *data = node->data;
    delete node;
    return data;
}

Data *List::removeFromEnd() {
    if (head == nullptr) return nullptr;

    Node *curr = head, *prev = nullptr;
    while (curr != tail) {
        prev = curr;
        curr = curr->next;
    }

    if (prev == nullptr) {
        head = tail = nullptr;
    }
    else {
        prev->next = nullptr;
        tail = prev;
    }

    Data *data = curr->data;
    delete curr;
    return data;
}

Data *List::removeFromIndex(int index) {
    if (head == nullptr) return nullptr;

    Node *curr = head, *prev = nullptr;
    int i = 0;
    while (i < index and curr != nullptr) {
        prev = curr;
        curr = curr->next;
    }

    if (curr == nullptr) return nullptr;

    if (prev == nullptr) head = curr->next;
    else prev->next = curr->next;

    if (tail == curr) {
        tail = prev;
    }

    Data* data = curr->data;
    delete curr;
    return data;
}

Data *List::getFromStart() {
    if (head == nullptr) return nullptr;
    return head->data;
}

Data *List::getFromEnd() {
    if (tail == nullptr) return nullptr;
    return tail->data;
}

Data *List::getFromIndex(int index) {
    if (head == nullptr) return nullptr;

    Node *curr = head;
    int i = 0;
    while (i < index and curr != nullptr) {
        curr = curr->next;
    }

    if (curr == nullptr) return nullptr;
    return curr->data;
}

bool List::isEmpty() {
    return head == nullptr;
}

List::~List() {
    Node *node = head;
    while (node != nullptr) {
        Node *tmp = node;
        node = node->next;
        delete tmp;
    }
}
